class Solution {
public:
    int minimumCost(vector<int>& cost) {
        int res = 0;
        int n = cost.size();
        sort(cost.begin(), cost.end(), greater<int>());

        for (int i = 1; i <= n; i++) {
            if (i % 3 != 0) {
                res = res + cost[i - 1];
            }
        }
        return res;
    }
};
