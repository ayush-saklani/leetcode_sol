class Solution {
public:
    int res = 0;
    int dfs(int node, int parent, vector<vector<int>>& adj, string& s) {
        int best1 = 0, best2 = 0;  // top two child paths
        for (int child : adj[node]) {
            if (child == parent) continue;
            int len = dfs(child, node, adj, s);
            if (s[child] == s[node]) continue;
            if (len > best1) { // just taking top 2 children from the parent node
                best2 = best1;
                best1 = len;
            } else if (len > best2) {
                best2 = len;
            }
        }
        res = max(res, best1 + best2 + 1); 
        return best1 + 1;
    }
    int longestPath(vector<int>& parent, string s) {
        int n = parent.size();
        vector<vector<int>> adj(n);
        for (int i = 1; i < n; i++) {
            adj[i].push_back(parent[i]);
            adj[parent[i]].push_back(i);
        }
        dfs(0, -1, adj, s);
        return res;
    }
};
// take 2 best children (+1 for parent) and update the res if greater 
// and put the best children length (+ 1 for parent) to continue dfs backtracking upward
