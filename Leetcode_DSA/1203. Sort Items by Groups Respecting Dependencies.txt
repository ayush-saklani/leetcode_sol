class Solution {
public:
    vector<int> sortItems(int n, int m, vector<int>& group, vector<vector<int>>& beforeItems) {
        // giving group to ungroup item as they need to be sort later on and number is required
        for(int i=0;i<n;i++){
            if(group[i] == -1){
                group[i] = m;
                m++;
            }
        }

        // step 1 - topo sort the items

        // making a graph of items according to before item for topo sort
        vector<vector<int>> adj(n);
        vector<int> indegree(n,0);  // calculating indegree on the fly
        for(int v=0;v<n;v++){
            for(auto& u: beforeItems[v]){
                adj[u].push_back(v);
                indegree[v]++;
            }
        }

        queue<int> q;
        for(int i=0;i<n;i++){
            if(indegree[i] == 0){
                q.push(i);
            }
        }

        vector<int> topo_item;
        while(!q.empty()){
            int front = q.front();
            topo_item.push_back(front);
            q.pop();
            for(auto i:adj[front]){
                indegree[i]--;
                if(indegree[i] == 0){
                    q.push(i);
                }
            }
        }
        if(topo_item.size()!=n) return {};  // cycle hai toh reject
        // topo_item is the topo sort of the items


        // step 2 - topo sort the groups
        // there are 0 to m-1 group 

        vector<vector<int>> group_adj(m);
        vector<int> group_indegree(m,0);
        vector<unordered_set<int>> seen(m);            // <--- dedup helper
        for(int i=0;i<n;i++){
            int item_group = group[i];
            for(int it:beforeItems[i]){
                int parent_group = group[it];
                if(item_group != parent_group  && !seen[parent_group].count(item_group)){
                    seen[parent_group].insert(item_group);
                    group_adj[parent_group].push_back(item_group);
                    group_indegree[item_group]++;
                }
            }
        }

        // queue<int> q; // previous q is empty so use that
        for(int i=0;i<m;i++){
            if(group_indegree[i] == 0){
                q.push(i);
            }
        }

        vector<int> topo_group;
        while(!q.empty()){
            int front = q.front();
            topo_group.push_back(front);
            q.pop();
            for(auto i:group_adj[front]){
                group_indegree[i]--;
                if(group_indegree[i] == 0){
                    q.push(i);
                }
            }
        }
        if (topo_group.size() != m) return {}; // cycle hai toh reject
        // topo_group is the topo sort of the items


        // step 3 - arrange all the items in topological sorted order in their respective group
        vector<vector<int>> items_in_group(m);
        for(int j=0;j<n;j++){
            items_in_group[group[topo_item[j]]].push_back(topo_item[j]);
        }

        // step 4 - add all items in the res from the groups (sorted in topological order).
        vector<int> res;
        for(int i = 0;i<m;i++){
            int g = topo_group[i];
            for(int item : items_in_group[g]){
                res.push_back(item);
            }
        }
        return res;
    }
};
